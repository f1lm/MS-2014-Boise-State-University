class Circle{ 
	double radius; 
	Circle(){ //无参数构造方法 
	} 
	Circle(double givenRadius){ 
		radius = givenRadius; 
	} 
	public void setRadius(double givenRadius){ 
		radius = givenRadius; 
	} 
	public double getRadius(){ 
		return radius; 
	} 
	public double computeArea(){ 
		return 3.1416*radius*radius; 
	} 
	public double computeRadiusFromGirth(int givenGirth){ 
		return givenGirth/3.14/2; //由周长计算半径 
	} 
} 
 
class Rectangle{ 
	double sideA, sideB; //边长 
	Rectangle(){ //无参数构造方法 
	} 
	Rectangle(double givenA, double givenB){ 
		sideA = givenA; 
		sideB = givenB; 
	} 
	public void setSide(double givenA, double givenB){ 
		sideA = givenA; 
		sideB = givenB; 
	} 
	public double getSideA(){ 
		return sideA; 
	} 
	public double getSideB(){ 
		return sideB; 
	} 
	public double computeArea(){ 
		return sideA*sideB; 
	} 
} 
 
public class TestObject{ 
	public static void main(String[] args){ 
		Circle circle = new Circle(); 
		//计算由两个百米弯道构成的圆的半径 
		double radius = circle.computeRadiusFromGirth(200); 
		circle.setRadius(radius); 
		Rectangle rect = new Rectangle(2*radius, 100); 
		double circleArea = circle.computeArea(); 
		double rectArea = rect.computeArea(); 
		double totalArea = circleArea + rectArea; 
		System.out.println("标准田径场的面积是："); 
		System.out.println(totalArea + "平方米"); 
	} 
}
