JavaScript plus Backbone


Description and History


Backbone.js was initially released on October of 2010.  Its developer, Jeremy Ashkenas, is also known for designing CoffeeScript, 
which adds syntactic sugar to enhance the brevity and readability of JavaScript.  Backbone is an open-source component of 
Document Cloud.  Version 1.1, a stable release of Backbone, was recently released on October 10, 2013.

Backbone is a light MVC framework that provides a means for organizing front-end codebases.  Web developers using JavaScript 
who find that their code suffers from inconsistency among its event handlers and/or lack of structure (aka "spaghetti code") 
would very likely benefit from the organization that Backbone structures give to JavaScript code.

Models
	Backbone models provide a place for a web application to store and validate client-side information, removing the 
	dependency on the DOM for data queries.
Views
	Without an organized approach to event handling in web application development, it is easy to produce inconsistent 
	event handlers in an application.  Backbone views provide a structure to organize all event handlers, promoting consistency
	and maintainability.
Collections
	With further complexity of a web application comes further opportunity to produce disorganized code.  Collections 
	organize multiple models into a single structure, which, again promotes structure and organization to a web application.

var ItemModel = Backbone.Model.extend({
	defaults: {
		name : "Jon Doe",
		age : 29,
		hobby : "Baseball",
		description : "A normal dude."
	},

	initialize: function() {
		console.log("New item model initialized.");
	}
});

var ItemView = Backbone.View.extend({
	tagName: "li",
	colorFlag: false,

	events: {
		"click" : "clickEvent"
	},

	template: _.template("<%= name %> - <%= age %> - <%= hobby %> - <%= description %>"),

	initialize: function() {
		this.render();
	},

	render: function() {
		this.$el.html(this.template(this.model.toJSON()));
		return this;
	},

	clickEvent: function() {
		if(this.colorFlag == false) {
			this.el.style.backgroundColor = "#00A6CF";
			this.colorFlag = true;
		}
		else {
			this.el.style.backgroundColor = "white";
			this.colorFlag = false;
		}
    }
});

var ItemCollectionView = Backbone.View.extend({
	tagName: "ul",
	id: "itemList",

	initialize: function() {
		this.render();
	},

	render: function () {
        var that = this;
        _.each(this.collection.models, function (item) {
            that.renderItemView(item);
        }, this);
	},

	renderItemView: function (item) {
        var itemView = new ItemView({
            model: item
        });
        $(this.el).append(itemView.el);
    },
});

var ItemCollection = Backbone.Collection.extend({
	model: ItemModel
});

function initItemExample1() {
	//init item models
	var im1 = new ItemModel();
	var im2 = new ItemModel({ name: "Chad Weigle", age: 21, hobby: "GAMING", description: "Sexy Beast!"});
	var im3 = new ItemModel({ name: "Thomas Green", age: 87, hobby: "Hermit", description: "Kinda old and tired." });
	var im4 = new ItemModel({ name: "William Unger", age: 20, hobby: "Ping Pong" });
	var im5 = new ItemModel({ name: "Melissa Clausen", age: 21, hobby: "Graphic Design, Painting, and Sketching", description: "A normal lady." });
	var im6 = new ItemModel({ name: "Kevin Philips", age: 87, hobby: "Hermit", description: "Kinda old and tired." });
	//init item collection
	var itemCollection = new ItemCollection();
	itemCollection.add(im1);
	itemCollection.add(im2);
	itemCollection.add(im3);
	itemCollection.add(im4);
	itemCollection.add(im5);
	itemCollection.add(im6);

	//Below is an easy way to declare a "global" variable within a function.
	//init item collection view
	window.itemCollectionView = new ItemCollectionView({collection: itemCollection});
}

function showItemExample1() {
	if ($("#itemListContainer").html().length == 0) {
		$("#itemListContainer").append(window.itemCollectionView.el);
	}
	else {
		$("#itemListContainer").toggle();
	}
}

DataView = Backbone.View.extend
({
	initialize: function()
	{
		var toAdd = $('<div class="todo_item"></div>');
		$('#list_container').append(toAdd);
		this.$el = toAdd;
		this.render();
	},
	render: function()
	{
		var variables = { lbl_description: $("#inp_toDoDescription").val() };
		$("#inp_toDoDescription").val("");
        var template = _.template( $("#list_template").html(), variables );
        this.$el.html( template );
	},
	events: 
	{
    	"click .btn_delete": "doDelete",
    	"click .btn_done" : "doDone",
    	"click .btn_up" : "doUp",
    	"click .btn_down" : "doDown"
	},
	doDelete: function( event )
	{
		this.remove();
	},
	doDone: function( event )
	{
		this.$el.find(".lbl_description").css("text-decoration","line-through")
	},
	doUp: function(event)
	{
		this.$el.insertBefore(this.$el.prev());
	},
	doDown: function(event)
	{
		this.$el.insertAfter(this.$el.next());
	}
});
